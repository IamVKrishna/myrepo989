## CREATING THE GITHUB REPO 

terraform {
  required_providers {
    github = {
      source  = "integrations/github"
      version = "~> 5.0"
    }
  }
}

provider "github" {
  token = "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"
}

# Create repo 

resource "github_repository" "r_name" {
  name        = "myrepo9740"
  description = "My awesome codebase"

  visibility = "public"

}
